98 #[Route('/charge-de-mission/liste-salles/details-salle/{nomsalle}', name: 'details_salle')]
99 public function details_salle(UserRepository $userRepository, JsonDataHandling $JsonDataHandling_service, ExperimentationRepository $experimentationRepository,SalleRepository $salleRepository, string $nomsalle): Response
100 {
101 // Salle inexistante ?
102 if ($salleRepository->findOneBy(['nom' => $nomsalle]) === null or !$experimentationRepository->estExistante($nomsalle)) {
103 return $this->redirectToRoute('liste_salles');
104 }
105
106 $etat_sa = $salleRepository->SAAssocie($nomsalle);
107
108 //extraction des dernière donnée d'une salle si il y en a pas alors est null
109 $dernieres_donnees = $JsonDataHandling_service->extraireDerniereDonneeSalle($nomsalle);
110
111 if($dernieres_donnees['date_de_capture'] != null){
112 $date_de_capture = new \DateTime($dernieres_donnees['date_de_capture']);
113 $now = new \DateTime();
114 $interval = $date_de_capture->diff($now);
115
116 // Format l'intervalle de temps de manière lisible
117 if ($interval->y > 0) {
118 $elapsed = $interval->y . ' années';
119 } elseif ($interval->m > 0) {
120 $elapsed = $interval->m . ' mois';
121 } elseif ($interval->d > 0) {
122 $elapsed = $interval->d . ' jours';
123 } elseif ($interval->h > 0) {
124 $elapsed = $interval->h . ' heures';
125 } elseif ($interval->i > 0) {
126 $elapsed = $interval->i . ' minutes';
127 } else {
128 $elapsed = $interval->s . ' secondes';
129 }
130 $intervalleTempSaison = $userRepository->intervallesTempSaison($dernieres_donnees['date_de_capture']);
131 }
132
133
134
135 //determine quel recommandation faire
136 $etatExp = $experimentationRepository->etatExp($nomsalle);
137 foreach ($etatExp as $exp) {
138 if ($exp['etat_exp'] == EtatExperimentation::demandeInstallation ) {
139 $recommandation = 'demande_installation_en_cours';
140 } elseif ($exp['etat_exp'] == EtatExperimentation::installee) {
141 $recommandation = 'installee';
142 } elseif ($exp['etat_exp'] == EtatExperimentation::demandeRetrait) {
143 $recommandation = 'demande_retrait_en_cours';
144 }
145 }
146 if(!isset($recommandation)){$recommandation = 'pas_de_exp';}
147
148
149 // Afficher la vue de salle details avec le résultat de l'existence
150 return $this->render('salle/details-salle.html.twig', [
151 //nom de la salle
152 'nomsalle' => $nomsalle,
153 //Infoemration sur le sa présent dans la salle si il existe
154 'etat_sa' => $etat_sa,
155 //dernière données de la salle, null si inexistantes
156 'dernieres_donnees' => $dernieres_donnees,
157 //temps écoulé depuis la dernière remonté de données
158 'elapsed' => $elapsed ?? null,
159 //liste d'une liste contenant des information sur l'intervalle et toutes les données associé, null si inexistantes
160 'recommandation' => $recommandation,
161
162 'intervalleTempSaison' => $intervalleTempSaison ?? null,
163 ]);
164 }
