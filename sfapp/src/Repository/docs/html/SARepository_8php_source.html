88 public function trierSA(array $resultat): array
89 {
90 foreach ($resultat as &$exp) {
91 if ($exp['exp_etat'] == EtatExperimentation::retiree) {
92 $exp['salle_nom'] = null;
93 }
94 }
95
96 $resultat = array_values($resultat);
97 $len = count($resultat);
98 for($i = 0;$i<$len;$i++){
99 $nom_sa = $resultat[$i]['sa_nom'];
100 $nombre_occurrences = 0;
101 foreach ($resultat as $element) {
102 if ($element['sa_nom'] === $nom_sa) {
103 $nombre_occurrences++;
104 }
105 }
106
107 if($resultat[$i]['exp_etat'] == EtatExperimentation::retiree and $resultat[$i]['salle_nom'] == null and $nombre_occurrences >= 2){
108 unset($resultat[$i]);
109 }
110 }
111 return $resultat;

160 public function filtrerSAGestionSA(array $etat = null, array $localisation = null): array
161 {
162 $exp = $this->toutLesSA();
163
164 if (!empty($etat) && $etat !== null) {
165 $etatsFiltres = array_map(function ($one) {
166 switch ($one) {
167 case 0:
168 return EtatSA::eteint;
169 case 1:
170 return EtatSA::marche;
171 case 2:
172 return EtatSA::probleme;
173 default:
174 return null;
175 }
176 }, $etat);
177
178 $exp = array_filter($exp, function ($item) use ($etatsFiltres) {
179 return in_array($item['sa_etat'], $etatsFiltres);
180 });
181 }
182
183 $exp = array_values($exp);
184
185 if (count($localisation) === 1 && $localisation !== null) {
186 if ($localisation[0] === 'salle') {
187 // Si $localisation est true, ajouter la condition pour salle.nom IS NOT NULL
188 $len = count($exp);
189 for($i = 0;$i<$len;$i++){
190 if($exp[$i]['exp_etat'] == EtatExperimentation::retiree or $exp[$i]['salle_nom'] == null){
191 unset($exp[$i]);
192 }
193 }
194 } elseif ($localisation[0] === 'stock') {
195 // Si $localisation est false, ajouter la condition pour salle.nom IS NULL
196 $len = count($exp);
197 for($i = 0;$i<$len;$i++){
198 if($exp[$i]['exp_etat'] == EtatExperimentation::installee or $exp[$i]['exp_etat'] == EtatExperimentation::demandeInstallation or $exp[$i]['exp_etat'] == EtatExperimentation::demandeRetrait or $exp[$i]['sa_etat'] != EtatSA::eteint){
199 unset($exp[$i]);
200 }
201 }
202 }
203 }
204
