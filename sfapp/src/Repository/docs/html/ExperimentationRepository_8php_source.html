486 public function triExperimentation(array $exp): array
487 {
488 usort($exp, function($a, $b) {
489 // Tri par nom_salle
490 if ($a['nom_salle'] != $b['nom_salle']) {
491 return $a['nom_salle'] <=> $b['nom_salle'];
492 }
493
494 // Si nom_salle est identique, tri par datedemande en ordre décroissant
495 return $b['datedemande'] <=> $a['datedemande'];
496 });
497
498 $len = count($exp);
499 if (count($exp) > 2) {
500 $salle = $exp[0]['nom_salle'];
501 for ($i = 0; $i < $len - 1; $i++) {
502 if ($salle == $exp[$i + 1]['nom_salle']) {
503 unset($exp[$i + 1]);
504 } else {
505 $salle = $exp[$i + 1]['nom_salle'];
506 }
507 }
508 }
509 return $exp;
510 }
